import discord
from discord import app_commands
from discord.ui import Button, View

class Cores:
    PRINCIPAL = 0x3498DB # Azul
    SUCESSO = 0x2ECC71    # Verde
    ERRO = 0xE74C3C       # Vermelho

class NovaEraBot(discord.Client):
    def __init__(self):
        super().__init__(intents=discord.Intents.all())
        self.synced = False

    async def on_ready(self):
        await self.wait_until_ready()
        if not self.synced:
            await tree.sync()
            self.synced = True
        print(f"🤖 {self.user.name} está online e pronto!")

client = NovaEraBot()
tree = app_commands.CommandTree(client)

FILA = []
LIMITE_FILA = 8  # Limite de 8 jogadores para o 4v4
FILA_ABERTA = False

class FilaInterface(View):
    def __init__(self):
        super().__init__(timeout=None)

    @discord.ui.button(label="Entrar na Fila", style=discord.ButtonStyle.green, custom_id="entrar_fila_btn")
    async def entrar(self, interaction: discord.Interaction, button: discord.ui.Button):
        global FILA, FILA_ABERTA
        
        if not FILA_ABERTA:
            await interaction.response.send_message("❌ A fila está fechada no momento!", ephemeral=True)
            return
            
        if interaction.user in FILA:
            await interaction.response.send_message("❌ Você já está na fila!", ephemeral=True)
            return

        if len(FILA) >= LIMITE_FILA:
            await interaction.response.send_message("❌ A fila já está cheia!", ephemeral=True)
            return

        FILA.append(interaction.user)
        embed = criar_embed_fila()
        await interaction.response.edit_message(embed=embed, view=self)

        if len(FILA) == LIMITE_FILA:
            mencoes = ", ".join([user.mention for user in FILA])
            embed_pronto = discord.Embed(
                title="🎮 FILA COMPLETA! JOGO PRONTO!",
                description=f"Os seguintes jogadores devem se preparar:\n\n{mencoes}\n\n*O Mediador irá liberar a sala!*",
                color=Cores.PRINCIPAL
            )
            await interaction.channel.send(content=mencoes, embed=embed_pronto)
            FILA.clear()
            embed_zerado = criar_embed_fila()
            await interaction.message.edit(embed=embed_zerado, view=self)

    @discord.ui.button(label="Sair da Fila", style=discord.ButtonStyle.danger, custom_id="sair_fila_btn")
    async def sair(self, interaction: discord.Interaction, button: discord.ui.Button):
        global FILA
        if interaction.user in FILA:
            FILA.remove(interaction.user)
            embed = criar_embed_fila()
            await interaction.response.edit_message(embed=embed, view=self)
        else:
            await interaction.response.send_message("❌ Você não está na fila.", ephemeral=True)

def criar_embed_fila():
    status = "🟢 ABERTA" if FILA_ABERTA else "🔴 FECHADA"
    lista_jogadores = "\n".join([f" `{i+1}°` - {user.mention}" for i, user in enumerate(FILA)]) if FILA else "*Nenhum jogador na fila.*"
    embed = discord.Embed(
        title="📝 FILA DE ESPA - NOVA ERA",
        description=f"Clique nos botões abaixo para participar.\n\n**Status da Fila:** {status}\n**Jogadores ({len(FILA)}/{LIMITE_FILA}):**\n{lista_jogadores}",
        color=Cores.PRINCIPAL
    )
    embed.set_footer(text="Nova Era Bot")
    return embed

@tree.command(name="painel", description="Envia o painel de controle da fila no canal")
@app_commands.checks.has_permissions(administrator=True)
async def enviar_painel(interaction: discord.Interaction):
    embed = criar_embed_fila()
    await interaction.response.send_message("Painel inicializado!", ephemeral=True)
    await interaction.channel.send(embed=embed, view=FilaInterface())

@tree.command(name="abrirfila", description="Abre as inscrições para a fila")
@app_commands.checks.has_permissions(manage_messages=True)
async def abrir_fila(interaction: discord.Interaction):
    global FILA_ABERTA
    FILA_ABERTA = True
    await interaction.response.send_message("🟢 A fila foi **ABERTA**!", ephemeral=False)

@tree.command(name="fecharfila", description="Fecha as inscrições da fila")
@app_commands.checks.has_permissions(manage_messages=True)
async def fechar_fila(interaction: discord.Interaction):
    global FILA_ABERTA
    FILA_ABERTA = False
    await interaction.response.send_message("🔴 A fila foi **FECHADA**!", ephemeral=False)
